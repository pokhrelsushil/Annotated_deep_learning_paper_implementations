# 
56    @staticmethod

to encourage the logits to be close to 0 

65        factor = 1e-4 / (y.shape[0] * y.shape[1])
66        maxx, ids = torch.max(y, -1, keepdim=True)
67        gy = torch.zeros_like(y)
68        gy.scatter_(-1, ids, maxx * factor)
